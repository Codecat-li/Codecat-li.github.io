数组的操作

//创建数组
let arr = [3,4,5]//另一种方式:let arr = new Array(1,2,3)

//常用的方法
arr.push(4)// 添加到最后一个元素
arr.pop(4) // 删除最后一个元素

arr.unshift(1)//添加到第一个元素
arr.shift()//删除第一个元素

arr.splice(0,2,4)//从第0个开始，删去2个元素,在删除的位置加一个元素：4

arr.sort((x, y)=>x - y)//数组自带的排序，x-y正序，y-x逆序。
